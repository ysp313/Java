/*
 * Classe Hello 
 * Permet de dire bonjour a condition de se présenter dans la ligne de commande
 */ 
 
 import java.util.Random; // J'ai besoin de l'outil Random qui sert à faire des tirages au sort
 
class Hello
{
	/*
	 * Cette méthode prend en parametre une chaine de caractère
	 * Elle renvoie "Hello " suivi de la valeur du parametre
	 * Si le parametre est null ou "" elle renvoie "Hello World"
	 */
	public static String hello(String firstName)
	{
		if (firstName != null && !firstName.equals(""))
		{
			return "Hello " + firstName;
		}
		return "Hello World";
	}
	
	
	/*
	 * Le programme principal, c'est le programme commence
	 */
	public static void main(String[] args)
	{
		Random generator = new Random(); // Je fabrique un générateur aléatoire
		
		//Je veux un entier entre [0 et X[ ou entre 0 inclus et X exclus
		int entier = generator.nextInt(45); 
		
		for(int i=0;i<10;i++)
		{
			entier = generator.nextInt(45);
			System.out.println(entier);
		}
		
		
		// Si il n'y a pas d'arguments dans la ligne de commande
		if(args.length == 0)
		{
			// On affiche hello world
			System.out.println(hello(null));
		}
		// Sinon on affiche autant de hello que d'arguments
		for(String firstName : args)
		{
			System.out.println(hello(firstName));
		}
​

		System.out.println(hello("")); // J'attends Hello World
		System.out.println("Je salue avec null");
		System.out.println(hello(null)); // J'attends Hello World
		* */
	}
}
